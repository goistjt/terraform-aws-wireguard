#cloud-config
package_update: true
package_upgrade: true

write_files:
  - path: /etc/wireguard/wg0.conf
    owner: root:root
    permissions: '0600'
    content: |
      [Interface]
      Address = ${wg_server_net}
      PrivateKey = ${wg_server_private_key}
      ListenPort = ${wg_server_port}
      PostUp   = iptables -A FORWARD -i %i -j ACCEPT; iptables -A FORWARD -o %i -j ACCEPT; iptables -t nat -A POSTROUTING -o ${wg_server_interface} -j MASQUERADE
      PostDown = iptables -D FORWARD -i %i -j ACCEPT; iptables -D FORWARD -o %i -j ACCEPT; iptables -t nat -D POSTROUTING -o ${wg_server_interface}

${peers}

  - path: /etc/wireguard/wg-bootstrap.sh
    owner: root:root
    permissions: '0700'
    content: |
      #!/bin/bash
      set -eux

      # Detect Ubuntu release
      RELEASE=$(lsb_release -rs)

      # Install packages
      if [ "$RELEASE" = "20.04" ]; then
        apt-get update -y
        apt-get install -y wireguard-dkms wireguard-tools awscli net-tools
      else
        # For 22.04/24.04
        apt-get update -y
        apt-get install -y wireguard-tools unzip curl net-tools

        # Install AWS CLI v2 if not already present
        if ! command -v aws >/dev/null 2>&1 || [[ $(aws --version 2>&1) != aws-cli/2* ]]; then
          ARCH=$(uname -m)
          if [ "$ARCH" = "aarch64" ]; then
            AWSCLI_ZIP="https://awscli.amazonaws.com/awscli-exe-linux-aarch64.zip"
          else
            AWSCLI_ZIP="https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip"
          fi

          TMP_DIR=$(mktemp -d)
          curl "$AWSCLI_ZIP" -o "$TMP_DIR/awscliv2.zip"
          unzip "$TMP_DIR/awscliv2.zip" -d "$TMP_DIR"
          "$TMP_DIR/aws/install" --bin-dir /usr/local/bin --install-dir /usr/local/aws-cli
          rm -rf "$TMP_DIR"
        fi
      fi

      # Associate Elastic IP if requested
      if [ "${use_eip}" != "disabled" ]; then
        # Get IMDSv2 token
        TOKEN=$(curl -sX PUT "http://169.254.169.254/latest/api/token" \
                  -H "X-aws-ec2-metadata-token-ttl-seconds: 21600")

        INSTANCE_ID=$(curl -s -H "X-aws-ec2-metadata-token: $TOKEN" \
                           http://169.254.169.254/latest/meta-data/instance-id)
        REGION=$(curl -s -H "X-aws-ec2-metadata-token: $TOKEN" \
                       http://169.254.169.254/latest/meta-data/placement/availability-zone | sed 's/[a-z]$//')

        if [ -n "$INSTANCE_ID" ] && [ -n "$REGION" ]; then
          aws --region "$REGION" ec2 associate-address \
              --allocation-id "${eip_id}" \
              --instance-id "$INSTANCE_ID"
        fi
      fi

      # Networking and MTU
      ip link set dev ${wg_server_interface} mtu 1420
      sed -i 's/#net.ipv4.ip_forward=1/net.ipv4.ip_forward=1/' /etc/sysctl.conf
      sysctl -p

      sysctl -w net.ipv4.ip_forward=1
      echo "net.ipv4.ip_forward=1" | tee /etc/sysctl.d/99-wireguard.conf
      sysctl --system

      iptables -F
      iptables -t nat -F
      iptables -X

      iptables -A INPUT -i ${wg_server_interface} -p udp --dport 51820 -j ACCEPT
      iptables -A FORWARD -i wg0 -o ${wg_server_interface} -s ${wg_server_net} -j ACCEPT
      iptables -A FORWARD -i ${wg_server_interface} -o wg0 -d ${wg_server_net} -m conntrack --ctstate RELATED,ESTABLISHED -j ACCEPT
      iptables -t nat -A POSTROUTING -s ${wg_server_net} -o ${wg_server_interface} -j MASQUERADE

      iptables -L -n -v
      iptables -t nat -L -n -v

      # Firewall rules
      ufw --force disable

      # Enable and start WireGuard
      systemctl enable wg-quick@wg0.service
      systemctl start wg-quick@wg0.service

runcmd:
  - /etc/wireguard/wg-bootstrap.sh
